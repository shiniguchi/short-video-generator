**waitlist.html.j2** — Replace the "Valid submission" block in the `<script>` section. The current code after spam checks simply hides the form and shows success. Replace with a fetch() call to POST /waitlist.

Find this block inside the submit handler (after the honeypot and timestamp checks):

```javascript
// Valid submission - show success message
// Note: Phase 16 will add backend processing
form.style.display = 'none';
successMessage.style.display = 'block';

console.log('Form submitted successfully');
return false;
```

Replace it with:

```javascript
// Get lp_source from meta tag (set by generator)
var lpSource = '';
var lpMeta = document.querySelector('meta[name="lp-source"]');
if (lpMeta) { lpSource = lpMeta.getAttribute('content') || ''; }

// Get API base URL (default: current origin for deployed LPs)
var apiBase = document.querySelector('meta[name="api-base"]');
var baseUrl = apiBase ? apiBase.getAttribute('content') : '';

var submitBtn = form.querySelector('button[type="submit"]');
var originalText = submitBtn.textContent;
submitBtn.disabled = true;
submitBtn.textContent = 'Submitting...';

// Error display element (reuse or create)
var errorEl = document.getElementById('form-error');
if (!errorEl) {
    errorEl = document.createElement('div');
    errorEl.id = 'form-error';
    errorEl.style.cssText = 'display:none;padding:0.75rem;background:#fef2f2;color:#991b1b;border-radius:8px;margin-bottom:1rem;font-size:0.875rem;';
    form.parentNode.insertBefore(errorEl, form.nextSibling);
}
errorEl.style.display = 'none';

fetch(baseUrl + '/waitlist', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ email: document.getElementById('email').value, lp_source: lpSource })
})
.then(function(res) {
    if (res.ok) { return res.json(); }
    return res.json().then(function(data) { throw data; });
})
.then(function(data) {
    form.style.display = 'none';
    successMessage.style.display = 'block';
})
.catch(function(err) {
    if (err && err.detail) {
        // Server returned structured error (409 duplicate, etc.)
        errorEl.textContent = err.detail;
        errorEl.style.display = 'block';
    } else {
        // Network error — show success anyway (offline graceful degradation)
        form.style.display = 'none';
        successMessage.style.display = 'block';
    }
})
.finally(function() {
    submitBtn.disabled = false;
    submitBtn.textContent = originalText;
});

return false;
```

Key design decisions:
- Reads `lp_source` from `<meta name="lp-source">` (set by base template) — no Jinja2 in section template.
- Reads optional `api-base` from meta tag for flexibility (empty = same origin for deployed LPs; can be set to `http://localhost:8000` for local dev).
- On network error, falls back to showing success (graceful offline degradation).
- On 409 (duplicate), shows server's friendly error message.
- Uses `var` not `const`/`let` for broader browser compatibility (matches LP target audience).
- Disables submit button during request to prevent double-submission.
- Dynamically creates error display element if not in DOM.
